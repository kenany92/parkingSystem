3   import org.apache.logging.log4j.LogManager;
4   import org.apache.logging.log4j.Logger;

6   import java.util.Scanner;

8   /**
9    * This class is dedicated to the input readers
10   * @author Nabil Boudjellal
11   */
12  public class InputReaderUtil {
13  
14      private static final Scanner scan = new Scanner(System.in,"UTF-8");
15      private static final Logger logger = LogManager.getLogger("InputReaderUtil");
16      /**
17       * This method scan and return the an int of the console entry
18       * @return int
19       */
20      public int readSelection() {
21          try {
22              return Integer.parseInt(scan.nextLine());
23          } catch(Exception e){
24              logger.error("Error while reading user input from Shell", e);
25              System.out.println("Error reading input. Please enter valid number for proceeding further");
26              return -1;
27          }
28      }
29      /**
30       * This method scan and return the a String of the console entry
31       * @return String
32       */
33      public String readVehicleRegistrationNumber() {
34          try {
35              String vehicleRegNumber= scan.nextLine();
36              if(vehicleRegNumber == null || vehicleRegNumber.trim().length()==0) {
37                  throw new IllegalArgumentException("Invalid input provided");
38              }
39              return vehicleRegNumber;
40          }catch(Exception e){
41              logger.error("Error while reading user input from Shell", e);
42              System.out.println("Error reading input. Please enter a valid string for vehicle registration number");
43              throw e;
44          }
45      }
